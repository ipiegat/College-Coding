\section*{Problem 2:}

\subsection*{2a.}
\[
\rho = \frac{\lambda}{\mu} = \frac{20}{30} = 0.6667
\]
\[
\rho = 0.6667 \, \text{(or 66.67\%)}
\]

\subsection*{2b.}
\[
L = \frac{\rho}{1 - \rho} = \frac{0.6667}{1 - 0.6667} = 2
\]

\subsection*{2c.}
\[
L_q = \rho \cdot L = 0.6667 \cdot 2 = 1.3334
\]

\subsection*{2d.}
